330 template <class RET>
331 void Context<RET>::validateTaskType(ITask::Type type) const
332 {
333  if (!_task)
334  {
335  throw std::runtime_error("Invalid task pointer");
336  }
337 
338  bool isValid = true;
339  switch (type)
340  {
341  case ITask::Type::Continuation:
342  case ITask::Type::ErrorHandler:
343  if ((_task->getType() != ITask::Type::First) &&
344  (_task->getType() != ITask::Type::Continuation))
345  {
346  isValid = false;
347  }
348  break;
349  case ITask::Type::Final:
350  if ((_task->getType() != ITask::Type::First) &&
351  (_task->getType() != ITask::Type::Continuation) &&
352  (_task->getType() != ITask::Type::ErrorHandler))
353  {
354  isValid = false;
355  }
356  break;
357  case ITask::Type::Termination:
358  if ((_task->getType() != ITask::Type::First) &&
359  (_task->getType() != ITask::Type::Continuation) &&
360  (_task->getType() != ITask::Type::ErrorHandler) &&
361  (_task->getType() != ITask::Type::Final))
362  {
363  isValid = false;
364  }
365  break;
366  default:
367  break;
368  }
369 
370  if (!isValid)
371  {
372  throw std::runtime_error("Restricted continuation method");
373  }
374 }

422 template <class RET>
423 void Context<RET>::sleep(std::chrono::milliseconds timeMs)
424 {
425  auto start = std::chrono::high_resolution_clock::now();
426  //wait until signalled or times out
427  while (1)
428  {
429  yield();
430  auto elapsed = std::chrono::high_resolution_clock::now() - start;
431  if (std::chrono::duration_cast<std::chrono::milliseconds>(elapsed) >= timeMs)
432  {
433  break; //timeout expired
434  }
435  }
436 }

438 template <class RET>
439 template <class OTHER_RET, class FUNC, class ... ARGS>
440 typename Context<OTHER_RET>::Ptr
441 Context<RET>::thenImpl(ITask::Type type, FUNC&& func, ARGS&&... args)
442 {
443  auto ctx = typename Context<OTHER_RET>::Ptr(new Context<OTHER_RET>(*this),
444  Context<OTHER_RET>::Delete);
445  auto task = Task::Ptr(new Task(ctx,
446  _task->getQueueId(), //keep current queueId
447  _task->isHighPriority(), //keep current priority
448  type,
449  std::forward<FUNC>(func),
450  std::forward<ARGS>(args)...),
451  Task::Delete);
452  ctx->setTask(task);
453 
454  //Chain tasks
455  std::static_pointer_cast<ITaskContinuation>(_task)->setNextTask(task);
456  task->setPrevTask(std::static_pointer_cast<ITaskContinuation>(_task));
457  return ctx;
458 }

780 Context<RET>::postImpl(int queueId, bool isHighPriority, ITask::Type type, FUNC&& func, ARGS&&... args)
781 {

783  {
784  throw std::runtime_error("Invalid queue id");
785  }

787  Context<OTHER_RET>::Delete);
788  auto task = Task::Ptr(new Task(ctx,
789  (queueId == (int)IQueue::QueueId::Same) ? _task->getQueueId() : queueId,
790  isHighPriority,
791  type,
792  std::forward<FUNC>(func),
793  std::forward<ARGS>(args)...),
794  Task::Delete);
795  ctx->setTask(task);
796  if (type == ITask::Type::Standalone)
797  {
798  _dispatcher->post(task);
799  }
