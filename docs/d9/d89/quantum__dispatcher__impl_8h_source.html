150 template <class RET, class FUNC, class ... ARGS>
151 typename ThreadContext<RET>::Ptr
152 TaskDispatcher::postImpl(int queueId,
153  bool isHighPriority,
154  ITask::Type type,
155  FUNC&& func,
156  ARGS&&... args)
157 {
158  if (_drain)
159  {
160  throw std::runtime_error("Posting is disabled");
161  }
162  if (queueId < (int)IQueue::QueueId::Any)
163  {
164  throw std::runtime_error("Invalid queue id");
165  }
166  auto ctx = typename Context<RET>::Ptr(new Context<RET>(_dispatcher),
167  Context<RET>::deleter);
168  auto task = Task::Ptr(new Task(ctx,
169  queueId,
170  isHighPriority,
171  type,
172  std::forward<FUNC>(func),
173  std::forward<ARGS>(args)...),
174  Task::deleter);
175  ctx->setTask(task);
176  if (type == ITask::Type::Standalone)
177  {
178  _dispatcher.post(task);
179  }
180  return std::static_pointer_cast<IThreadContext<RET>>(ctx);
181 }

183 template <class RET, class FUNC, class ... ARGS>
184 typename ThreadFuture<RET>::Ptr
185 TaskDispatcher::postAsyncIoImpl(int queueId,
186  bool isHighPriority,
187  FUNC&& func,
188  ARGS&&... args)
189 {
190  if (_drain)
191  {
192  throw std::runtime_error("Posting is disabled");
193  }
194  if (queueId < (int)IQueue::QueueId::Any)
195  {
196  throw std::runtime_error("Invalid queue id");
197  }
198  auto promise = typename Promise<RET>::Ptr(new Promise<RET>(), Promise<RET>::deleter);
199  auto task = IoTask::Ptr(new IoTask(promise,
200  queueId,
201  isHighPriority,
202  std::forward<FUNC>(func),
203  std::forward<ARGS>(args)...),
204  IoTask::deleter);
205  _dispatcher.postAsyncIo(task);
206  return promise->getIThreadFuture();
207 }
