67 template <typename T, unsigned int SIZE>
68 typename StackAllocator<T,SIZE>::pointer StackAllocator<T,SIZE>::allocate(size_type,
69  std::allocator<void>::const_pointer)
70 {
71  bool useHeap = false;
72  {
73  SpinLock::Guard lock(_spinlock);
74  useHeap = isEmpty();
75  if (!isEmpty())
76  {
77  return reinterpret_cast<pointer>(&_buffer[_freeBlocks[_freeBlockIndex--]]);
78  }
79  }
80  if (useHeap)
81  {
82  ++_numHeapAllocatedBlocks;
83  return (pointer)new char[sizeof(value_type)];
84  }
85  return nullptr; //never reaches here
86 }

88 template <typename T, unsigned int SIZE>
89 void StackAllocator<T,SIZE>::deallocate(pointer p, size_type)
90 {
91  if (p == nullptr) {
92  return;
93  }
94  if (isStack(p)) {
95  //find index of the block
96  SpinLock::Guard lock(_spinlock);
97  _freeBlocks[++_freeBlockIndex] = blockIndex(p);
98  }
99  else {
100  delete[](char*)p;
101  --_numHeapAllocatedBlocks;
102  assert(_numHeapAllocatedBlocks >= 0);
103  }
104 }
